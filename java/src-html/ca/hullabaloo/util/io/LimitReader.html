001    package ca.hullabaloo.util.io;
002    
003    import java.io.IOException;
004    import java.io.Reader;
005    
006    /**
007     * Limits the number of characters returned from this reader -- does not account
008     * for {@link Reader#skip(long)}
009     * 
010     * @author Darren Gilroy
011     */
012    public class LimitReader extends CountingReader {
013            private final long limit;
014    
015            public LimitReader(Reader reader, long limit) {
016                    super(reader);
017                    this.limit = limit;
018    
019                    if (this.limit < 0)
020                            throw new IllegalArgumentException("Limit must be positive");
021            }
022    
023            @Override
024            public int read() throws IOException {
025                    if (getCount() >= limit)
026                            return -1;
027                    else
028                            return super.read();
029            }
030    
031            @Override
032            public int read(char[] ch, int start, int len) throws IOException {
033                    if (getCount() >= limit)
034                            return -1;
035    
036                    // calcualate what our count will be after this method is called,
037                    // and possibly reduce the amount of data we're asking for if
038                    // it would be more than our limit.
039                    long count = len + getCount();
040                    if (count > limit)
041                            len -= (count - limit);
042    
043                    return super.read(ch, start, len);
044            }
045    
046            @Override
047            public int read(char[] ch) throws IOException {
048                    return this.read(ch, 0, ch.length);
049            }
050    }
