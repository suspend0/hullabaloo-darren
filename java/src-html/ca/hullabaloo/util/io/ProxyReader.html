001    package ca.hullabaloo.util.io;
002    
003    import java.io.IOException;
004    import java.io.Reader;
005    import java.nio.CharBuffer;
006    
007    /**
008     * Deprecated: Use IOUtils ProxyReader
009     */
010    public class ProxyReader extends Reader {
011            private final Reader delegate;
012    
013            /**
014             * Create a DelegatingReader with the given delegate
015             * 
016             * @param delegate
017             */
018            public ProxyReader(final Reader delegate) {
019                    this.delegate = delegate;
020            }
021    
022            /**
023             * @see java.io.Reader#close()
024             */
025            @Override
026            public void close() throws IOException {
027                    this.delegate.close();
028            }
029    
030            /**
031             * @return Returns the delegate.
032             */
033            public Reader getDelegate() {
034                    return this.delegate;
035            }
036    
037            /**
038             * @see java.io.Reader#mark(int)
039             */
040            @Override
041            public void mark(int readAheadLimit) throws IOException {
042                    this.delegate.mark(readAheadLimit);
043            }
044    
045            /**
046             * @see java.io.Reader#markSupported()
047             */
048            @Override
049            public boolean markSupported() {
050                    return this.delegate.markSupported();
051            }
052    
053            /**
054             * @see java.io.Reader#read()
055             */
056            @Override
057            public int read() throws IOException {
058                    return this.delegate.read();
059            }
060    
061            /**
062             * @see java.io.Reader#read(char[])
063             */
064            @Override
065            public int read(char[] cbuf) throws IOException {
066                    return this.delegate.read(cbuf);
067            }
068    
069            /**
070             * @see java.io.Reader#read(char[], int, int)
071             */
072            @Override
073            public int read(char[] cbuf, int off, int len) throws IOException {
074                    return this.delegate.read(cbuf, off, len);
075            }
076    
077            /**
078             * @see java.io.Reader#read(java.nio.CharBuffer)
079             */
080            @Override
081            public int read(CharBuffer target) throws IOException {
082                    return this.delegate.read(target);
083            }
084    
085            /**
086             * @see java.io.Reader#ready()
087             */
088            @Override
089            public boolean ready() throws IOException {
090                    return this.delegate.ready();
091            }
092    
093            /**
094             * @see java.io.Reader#reset()
095             */
096            @Override
097            public void reset() throws IOException {
098                    this.delegate.reset();
099            }
100    
101            /**
102             * @see java.io.Reader#skip(long)
103             */
104            @Override
105            public long skip(long n) throws IOException {
106                    return this.delegate.skip(n);
107            }
108    
109    }
