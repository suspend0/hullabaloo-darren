001    package ca.hullabaloo.util.text;
002    
003    /**
004     * Interface for things that want to know when a match occurs
005     */
006    public interface PhraseMatchListener {
007            /**
008             * Signals a match was found. Note that
009             * <code> input.subSequence(start, start +
010             * length).equals(phrase) </code>
011             * will be <code>true</code> for all usual cases. However, if the matcher
012             * is not case sensitive it might not be equal -- some case transformations
013             * cause the string to grow or shrink. See
014             * {@link String#toLowerCase(java.util.Locale)}for details.
015             * 
016             * @param input
017             *            the input we matched against
018             * @param start
019             *            the starting character of the match in the input string
020             * @param length
021             *            the length of the match
022             * @param phrase
023             *            the phrase we found
024             * @param wordNumber
025             *            the index of the starting <b>word </b> in the input string
026             * @param wordCount
027             *            the number of words in the phrase in the effective locale
028             */
029            public void matched(CharSequence input, int start, int length,
030                            CharSequence phrase, int wordNumber, int wordCount);
031    }
