001    package ca.hullabaloo.util;
002    
003    /**
004     * From http://java.sun.com/developer/JDCTechTips/2002/tt0604.html
005     */
006    public class ArrayCharSequence implements CharSequence {
007            private char[] vec;
008    
009            private int off;
010    
011            private int len;
012    
013            /** Construct a ArrayCharSequence from a char array */
014            public ArrayCharSequence() {
015                    // must set char array before use
016            }
017    
018            /** Construct a ArrayCharSequence from a char array */
019            public ArrayCharSequence(char[] ch) {
020                    this(ch, 0, ch.length);
021            }
022    
023            /** set the character array */
024            public void setCharArray(char[] ch) {
025                    setCharArray(ch, 0, ch.length);
026            }
027    
028            /** set the character array */
029            public void setCharArray(char[] ch, int start, int length) {
030                    if (ch == null) {
031                            throw new NullPointerException("null vec");
032                    }
033                    if (start < 0 || length < 0 || start > ch.length - length) {
034                            throw new IllegalArgumentException("bad off/len");
035                    }
036                    this.vec = ch;
037                    this.off = start;
038                    this.len = length;
039            }
040    
041            /** construct an CharArrayWrapper from a portion of a char array */
042            public ArrayCharSequence(char ch[], int start, int length) {
043                    setCharArray(ch, start, length);
044            }
045    
046            /** length of sequence */
047            public int length() {
048                    return len;
049            }
050    
051            /** character at a given index */
052            public char charAt(int index) {
053                    if (index < 0 || index >= len) {
054                            throw new IndexOutOfBoundsException("invalid index");
055                    }
056                    return vec[index + off];
057            }
058    
059            /** subsequence from start (inclusive) to end (exclusive) */
060            public CharSequence subSequence(int start, int end) {
061                    if (start < 0 || end < 0 || end > len || start > end) {
062                            throw new IndexOutOfBoundsException("invalid start/end");
063                    }
064                    return new ArrayCharSequence(vec, start + off, end - start);
065            }
066    
067            /** convert to string */
068            @Override
069            public String toString() {
070                    return new String(vec, off, len);
071            }
072    
073    }
