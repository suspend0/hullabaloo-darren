001    package ca.hullabaloo.util.transform;
002    
003    import java.net.URI;
004    import java.net.URISyntaxException;
005    
006    import javax.xml.transform.Source;
007    import javax.xml.transform.TransformerException;
008    import javax.xml.transform.URIResolver;
009    
010    /**
011     * Simply resolves the href against the base before passing to the delegate.
012     * Useful in {@link URIREsolverChain}s where it's wasteful to resolve the url
013     * each time.
014     * 
015     * @author darren
016     */
017    public class ResolvingURIResolver implements URIResolver {
018            private final URIResolver delegate;
019    
020            /** Creates a new ResolvingURIResolver, passing requests to the delegate */
021            public ResolvingURIResolver(URIResolver delegate) {
022                    if (delegate == null)
023                            throw new NullPointerException(
024                                            "Can not delegate to a null resolver");
025    
026                    this.delegate = delegate;
027            }
028    
029            /**
030             * Resolves the href against he base, and then passes
031             * <code>(resolved,null)</code> to the delegate.
032             */
033            public Source resolve(String href, String base) throws TransformerException {
034                    if (base == null) {
035                            return delegate.resolve(href, null);
036                    }
037    
038                    if (href.startsWith("fbf://") || href.startsWith("http://")
039                                    || href.startsWith("https://")) {
040                            return delegate.resolve(href, null);
041                    }
042    
043                    try {
044                            URI uri = new URI(base).resolve(href);
045                            return delegate.resolve(uri.toString(), null);
046    
047                            //
048                    } catch (URISyntaxException e) {
049                            throw new TransformerException("Error resolving " + href
050                                            + " against " + base, e);
051                    }
052    
053            }
054    
055            /** Returns the delegate */
056            public URIResolver getWrapped() {
057                    return this.delegate;
058            }
059    
060    }
