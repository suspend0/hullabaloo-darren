001    package ca.hullabaloo.util.transform;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    import java.util.logging.Logger;
006    
007    import javax.xml.transform.Source;
008    import javax.xml.transform.TransformerException;
009    import javax.xml.transform.URIResolver;
010    
011    /**
012     * A URI resolver that resolves a url from each resolver in series until a
013     * non-null result is returned.
014     * <p>
015     * This class is <b>not synchronized</b>, do not add resolvers when you are
016     * resolving with this class.
017     * </p>
018     * 
019     * @author Darren Gilroy
020     * @version $Revision$
021     */
022    public class URIResolverChain implements URIResolver {
023            // Logger
024            private static final Logger logger = Logger
025                            .getLogger(URIResolverChain.class.getName());
026    
027            /** No-op resolver */
028            private static final URIResolver NO_OP_RESOLVER = new URIResolver() {
029                    public Source resolve(String href, String base)
030                                    throws TransformerException {
031                            return null;
032                    }
033            };
034    
035            /** The resolver chain */
036            private final List<URIResolver> resolverChain = new ArrayList<URIResolver>();
037    
038            /** The resolver we resolve with */
039            private URIResolver resolver;
040    
041            /** Resolve the base before entering the chain */
042            private boolean resolveBase = true;
043    
044            /** If we're using the full implementation */
045            private boolean usingFullImpl = false;
046    
047            /**
048             * Creates a new URIResolverChain object.
049             */
050            public URIResolverChain() {
051                    // default resolver in case somebody tries to use it with no underlying
052                    // resolvers
053                    this.resolver = NO_OP_RESOLVER;
054            }
055    
056            /**
057             * Creates a new URIResolverChain object.
058             * 
059             * @param resolver1
060             */
061            public URIResolverChain(URIResolver resolver1) {
062                    if (resolver1 == null)
063                            throw new NullPointerException("Resolver1 is null");
064    
065                    this.resolverChain.add(resolver1);
066                    makeResolver();
067            }
068    
069            /**
070             * Creates a new URIResolverChain object.
071             * 
072             * @param resolver1
073             * @param resolver2
074             */
075            public URIResolverChain(URIResolver resolver1, URIResolver resolver2) {
076                    if (resolver1 == null)
077                            throw new NullPointerException("Resolver1 is null");
078                    if (resolver2 == null)
079                            throw new NullPointerException("Resolver2 is null");
080    
081                    this.resolverChain.add(resolver1);
082                    this.resolverChain.add(resolver2);
083                    makeResolver();
084            }
085    
086            /**
087             * Creates a new URIResolverChain object.
088             * 
089             * @param resolver1
090             * @param resolver2
091             * @param resolver3
092             */
093            public URIResolverChain(URIResolver resolver1, URIResolver resolver2,
094                            URIResolver resolver3) {
095                    if (resolver1 == null)
096                            throw new NullPointerException("Resolver1 is null");
097                    if (resolver2 == null)
098                            throw new NullPointerException("Resolver2 is null");
099                    if (resolver3 == null)
100                            throw new NullPointerException("Resolver3 is null");
101    
102                    this.resolverChain.add(resolver1);
103                    this.resolverChain.add(resolver2);
104                    this.resolverChain.add(resolver3);
105                    makeResolver();
106            }
107    
108            /**
109             * Adds a resolver to the chain.
110             * 
111             * @param resolver
112             *            the resolver to add
113             */
114            public void addResolver(final URIResolver resolver) {
115                    if (resolver == null)
116                            throw new NullPointerException("Resolver is null");
117    
118                    this.resolverChain.add(resolver);
119                    makeResolver();
120            }
121    
122            /** Clears the resolver chain */
123            public void clear() {
124                    this.resolverChain.clear();
125                    makeResolver();
126            }
127    
128            /** If we should use the resolver */
129            public void doResolveBase(boolean resolve) {
130                    this.resolveBase = resolve;
131            }
132    
133            /**
134             * Resolves using the chain
135             * 
136             * @param href
137             *            the href
138             * @param base
139             *            the base
140             * @return a Source, or null
141             * @throws TransformerException
142             *             problem
143             */
144            public Source resolve(String href, String base) throws TransformerException {
145                    return this.resolver.resolve(href, base);
146            }
147    
148            /** Makes the resolver from the current chain */
149            private void makeResolver() {
150                    if (this.usingFullImpl)
151                            return;
152    
153                    switch (this.resolverChain.size()) {
154                    case 0:
155                            this.resolver = NO_OP_RESOLVER;
156                            break;
157                    case 1:
158                            this.resolver = this.resolverChain.get(0);
159                            break;
160                    case 2:
161                            this.resolver = newChain(this.resolverChain.get(0),
162                                            this.resolverChain.get(1));
163                            break;
164                    case 3:
165                            this.resolver = newChain(this.resolverChain.get(0),
166                                            this.resolverChain.get(1), this.resolverChain.get(2));
167                            break;
168                    default:
169                            this.usingFullImpl = true;
170                            this.resolver = new URIResolverChainImpl(this.resolverChain);
171                    }
172    
173                    if (this.resolveBase)
174                            this.resolver = new ResolvingURIResolver(this.resolver);
175            }
176    
177            /**
178             * Returns a URIResolver (not necessarily an instance of URIResolverChain)
179             * optimized for two resolvers)
180             * 
181             * @param resolver1
182             *            the first resolver
183             * @param resolver2
184             *            the second resolver
185             * @return a resolver
186             */
187            private static URIResolver newChain(final URIResolver resolver1,
188                            final URIResolver resolver2) {
189                    return new URIResolver() {
190                            public Source resolve(String href, String base)
191                                            throws TransformerException {
192                                    logger.finer(href);
193    
194                                    Source r = null;
195    
196                                    if ((r = resolver1.resolve(href, base)) != null) {
197                                            return r;
198                                    }
199    
200                                    if ((r = resolver2.resolve(href, base)) != null) {
201                                            return r;
202                                    }
203    
204                                    logger.finer("Returning null");
205    
206                                    return null;
207                            }
208                    };
209            }
210    
211            /**
212             * Returns a URIResolver (not necessarily an instance of URIResolverChain)
213             * optimized for three resolvers)
214             * 
215             * @param resolver1
216             *            the first resolver
217             * @param resolver2
218             *            the second resolver
219             * @param resolver3
220             *            the third resolver
221             * @return a URIResolver
222             */
223            private static URIResolver newChain(final URIResolver resolver1,
224                            final URIResolver resolver2, final URIResolver resolver3) {
225                    return new URIResolver() {
226                            public Source resolve(String href, String base)
227                                            throws TransformerException {
228                                    logger.finer(href);
229    
230                                    Source r = null;
231    
232                                    if ((r = resolver1.resolve(href, base)) != null) {
233                                            return r;
234                                    }
235    
236                                    if ((r = resolver2.resolve(href, base)) != null) {
237                                            return r;
238                                    }
239    
240                                    if ((r = resolver3.resolve(href, base)) != null) {
241                                            return r;
242                                    }
243    
244                                    logger.finer("Returning null");
245    
246                                    return null;
247                            }
248                    };
249            }
250    
251            /**
252             * Class that uses a dynamic list of resolvers.
253             */
254            private static class URIResolverChainImpl implements URIResolver {
255                    private final List<URIResolver> chain;
256    
257                    /**
258                     * Creates a new URIResolverChainImpl object.
259                     * 
260                     * @param chain
261                     *            the list of resolvers to use.
262                     */
263                    URIResolverChainImpl(List<URIResolver> chain) {
264                            this.chain = chain;
265                    }
266    
267                    /**
268                     * Resolve the url
269                     * 
270                     * @param href
271                     *            the href
272                     * @param base
273                     *            the base
274                     * @return a source, or null
275                     * @throws TransformerException
276                     *             problem
277                     */
278                    public Source resolve(String href, String base)
279                                    throws TransformerException {
280                            final int j = this.chain.size();
281    
282                            for (int i = 0; i < j; i++) {
283                                    Source r = this.chain.get(i).resolve(href, base);
284    
285                                    if (r != null) {
286                                            return r;
287                                    }
288                            }
289    
290                            return null;
291                    }
292            }
293    }
