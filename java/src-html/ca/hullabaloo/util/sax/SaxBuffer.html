035    /**
036     * A class that can record SAX events and replay them later.
037     * 
038     * <p>
039     * Use this class if you need to frequently generate smaller amounts of SAX
040     * events, or replay a set of recorded start events immediately.
041     * </p>
042     * 
043     * <p>
044     * Both ContentHandler and LexicalHandler are supported, the only exception is
045     * that the setDocumentLocator event is not recorded.
046     * </p>
047     * 
048     * <p>
049     * The original package for this class is <code>org.apache.cocoon.xml</code>
050     * </p>
051     * 
052     * <p>
053     * Note this class is not strictly thread-safe, although once the buffer is
054     * built, you can safely call toSAX() from multiple threads (as long as there
055     * are no concurrent modifications).
056     * </p>
057     * 
058     * @author <a href="mailto:dev@cocoon.apache.org">Apache Cocoon Team</a>
059     * @author Darren Gilroy
060     */
061    public class SaxBuffer implements ContentHandler, LexicalHandler {
062            /** Stores list of {@link SaxBit} objects. */
063            protected List<SaxBit> saxbits = new ArrayList<SaxBit>(100);
064    
065            /**
066             * Creates empty SaxBuffer
067             */
068            public SaxBuffer() {
069            }
070    
071            /**
072             * Creates copy of another SaxBuffer
073             * 
074             * @param saxBuffer
075             */
076            public SaxBuffer(SaxBuffer saxBuffer) {
077                    append(saxBuffer);
078            }
079    
080            /**
081             * @see ContentHandler#setDocumentLocator(org.xml.sax.Locator)
082             */
083            public void setDocumentLocator(Locator locator) {
084                    // don't record this event
085            }
086    
087            /**
088             * Returns true if this SaxBuffer is emmpty.
089             * 
090             * @return true if empty
091             */
092            public boolean isEmpty() {
093                    return saxbits.isEmpty();
094            }
095    
096            /**
097             * Appends the bits in the SaxBuffer to this Buffer. Much the same as
098             * otherBuffer.toSax(thisBuffer), except doesn't create new copies of the
099             * bits objects.
100             * 
101             * @param saxBuffer
102             *            the buffer to append
103             */
104            public void append(SaxBuffer saxBuffer) {
105                    this.saxbits.addAll(saxBuffer.saxbits);
106            }

246            /**
247             * Sends the events to the content handler
248             * 
249             * @param contentHandler
250             *            the destination
251             * 
252             * @throws SAXException
253             *             from the content handler
254             */
255            public void toSAX(ContentHandler contentHandler) throws SAXException {
256                    if (contentHandler instanceof LexicalHandler)
257                            toSAX(contentHandler, (LexicalHandler) contentHandler);
258                    else
259                            toSAX(contentHandler, SAXUtils.DEFAULT_HANDLER);
260            }
261    
262            /**
263             * Sends the events to the content handler
264             * 
265             * @param contentHandler
266             *            the destinations
267             * 
268             * @throws SAXException
269             *             from the content handler
270             */
271            public void toSAX(ContentHandler contentHandler,
272                            LexicalHandler lexicalHandler) throws SAXException {
273                    for (Iterator i = saxbits.iterator(); i.hasNext();) {
274                            SaxBit saxbit = (SaxBit) i.next();
275                            saxbit.send(contentHandler, lexicalHandler);
276                    }
277            }
278    
279            /**
280             * Used in i18n XML bundle implementation
281             * 
282             * @return string
283             */
284            public String toString() {
285                    StringBuffer value = new StringBuffer();
286    
287                    for (Iterator i = saxbits.iterator(); i.hasNext();) {
288                            SaxBit saxbit = (SaxBit) i.next();
289    
290                            if (saxbit instanceof Characters) {
291                                    ((Characters) saxbit).toString(value);
292                            }
293                    }
294    
295                    return value.toString();
296            }

298            /**
299             * Adds a SaxBit to the bits list
300             * 
301             * @param bit -
302             */
303            protected final void addBit(SaxBit bit) {
304                    saxbits.add(bit);
305            }

307            /**
308             * Iterates through the bits list
309             * 
310             * @return Iterator
311             */
312            protected final Iterator bits() {
313                    return saxbits.iterator();
314            }

316            /**
317             * SaxBit is a representation of the SAX event. Every SaxBit is immutable
318             * object.
319             */
320            interface SaxBit {
321                    /**
322                     * Replay the event to the ContentHandler or lexical handler
323                     * 
324                     * @param contentHandler
325                     *            target
326                     * 
327                     * @throws SAXException
328                     *             api
329                     */
330                    public void send(ContentHandler contentHandler,
331                                    LexicalHandler lexicalHandler) throws SAXException;
332    
333            }
334    
335            /**
336             * Represents a SAX Event
337             */
338            static final class Characters implements SaxBit {
339                    private final char[] ch;
340    
341                    /**
342                     * Creates a new Characters object.
343                     * 
344                     * @param ch
345                     * @param start
346                     * @param length
347                     */
348                    public Characters(char[] ch, int start, int length) {
349                            // make a copy so that we don't hold references to a potentially
350                            // large array we don't control
351                            this.ch = new char[length];
352                            System.arraycopy(ch, start, this.ch, 0, length);
353                    }
354    
355                    public void send(ContentHandler contentHandler,
356                                    LexicalHandler lexicalHandler) throws SAXException {
357                            contentHandler.characters(ch, 0, ch.length);
358                    }
359    
360                    /**
361                     * Appends the characters to the passed stringbuffer
362                     * 
363                     * @param value
364                     *            the buffer to append to.
365                     */
366                    public void toString(StringBuffer value) {
367                            value.append(ch);
368                    }
369            }
