001    package ca.hullabaloo.util.sax;
002    
003    import org.xml.sax.Attributes;
004    import org.xml.sax.ContentHandler;
005    import org.xml.sax.Locator;
006    import org.xml.sax.SAXException;
007    
008    /**
009     * Duplicates content handler events to two content handlers.
010     * 
011     * @author Darren Gilroy
012     */
013    public class TeeContentHandler implements ContentHandler {
014            protected final ContentHandler ch;
015    
016            protected final ContentHandler branch;
017    
018            public TeeContentHandler(ContentHandler ch, ContentHandler branch) {
019                    this.ch = ch;
020                    this.branch = branch;
021            }
022    
023            public void characters(char[] ch, int start, int length)
024                            throws SAXException {
025                    this.ch.characters(ch, start, length);
026                    this.branch.characters(ch, start, length);
027            }
028    
029            public void endDocument() throws SAXException {
030                    this.ch.endDocument();
031                    this.branch.endDocument();
032            }
033    
034            public void endElement(String namespace, String localName, String qName)
035                            throws SAXException {
036                    this.ch.endElement(namespace, localName, qName);
037                    this.branch.endElement(namespace, localName, qName);
038            }
039    
040            public void endPrefixMapping(String prefix) throws SAXException {
041                    this.ch.endPrefixMapping(prefix);
042                    this.branch.endPrefixMapping(prefix);
043            }
044    
045            public void ignorableWhitespace(char[] ch, int start, int length)
046                            throws SAXException {
047                    this.ch.ignorableWhitespace(ch, start, length);
048                    this.branch.ignorableWhitespace(ch, start, length);
049            }
050    
051            public void processingInstruction(String target, String data)
052                            throws SAXException {
053                    this.ch.processingInstruction(target, data);
054                    this.branch.processingInstruction(target, data);
055            }
056    
057            public void setDocumentLocator(Locator locator) {
058                    this.ch.setDocumentLocator(locator);
059                    this.branch.setDocumentLocator(locator);
060            }
061    
062            public void skippedEntity(String name) throws SAXException {
063                    this.ch.skippedEntity(name);
064                    this.branch.skippedEntity(name);
065            }
066    
067            public void startDocument() throws SAXException {
068                    this.ch.startDocument();
069                    this.branch.startDocument();
070            }
071    
072            public void startElement(String namespace, String localname, String qName,
073                            Attributes atts) throws SAXException {
074                    this.ch.startElement(namespace, localname, qName, atts);
075                    this.branch.startElement(namespace, localname, qName, atts);
076            }
077    
078            public void startPrefixMapping(String prefix, String uri)
079                            throws SAXException {
080                    this.ch.startPrefixMapping(prefix, uri);
081                    this.branch.startPrefixMapping(prefix, uri);
082            }
083    
084    }
