001    package ca.hullabaloo.util.sax;
002    
003    import org.xml.sax.Attributes;
004    import org.xml.sax.ContentHandler;
005    import org.xml.sax.Locator;
006    import org.xml.sax.SAXException;
007    import org.xml.sax.SAXParseException;
008    
009    /**
010     * Content Handler that passes the events it receives to a parent content
011     * handler.
012     * 
013     * @author Darren Gilroy
014     */
015    public class ProxyContentHandler implements ContentHandler {
016            /** the one we send our events too */
017            private ContentHandler m_parent;
018    
019            /** Locator for parse exceptions */
020            private Locator m_locator;
021    
022            /**
023             * Creates a new ProxyContentHandler object.
024             * 
025             * @param handler
026             *            The parent content handler to send non-transform elements too
027             */
028            public ProxyContentHandler(ContentHandler handler) {
029                    // handler
030                    setParent(handler);
031            }
032    
033            /**
034             * Creates a new ProxyContentHandler object.
035             */
036            public ProxyContentHandler() {
037            }
038    
039            /**
040             * save my locator so I can throw my own exceptions, and then proxies this
041             * call to the parent content handler
042             * 
043             * @param locator
044             *            {@link ContentHandler}
045             */
046            public void setDocumentLocator(Locator locator) {
047                    m_locator = locator;
048                    m_parent.setDocumentLocator(locator);
049            }
050    
051            /**
052             * Returns the document locator so you can create your own
053             * SAXParseExceptions
054             * 
055             * @see #parseException(String)
056             */
057            public Locator getDocumentLocator() {
058                    return m_locator;
059            }
060    
061            /**
062             * sets the parenet content handler
063             * 
064             * @param ch
065             *            the new parent
066             */
067            public void setParent(ContentHandler ch) {
068                    m_parent = ch;
069            }
070    
071            /**
072             * gets the parent content handler
073             * 
074             * @return Content Handler
075             */
076            public ContentHandler getParent() {
077                    return m_parent;
078            }
079    
080            /**
081             * Proxies this call to the parent content handler
082             * 
083             * @param ch
084             *            See {@link ContentHandler}
085             * @param start
086             *            See {@link ContentHandler}
087             * @param length
088             *            See {@link ContentHandler}
089             * 
090             * @throws SAXException
091             *             See {@link ContentHandler}
092             */
093            public void characters(char[] ch, int start, int length)
094                            throws SAXException {
095                    m_parent.characters(ch, start, length);
096            }
097    
098            /**
099             * Proxies this call to the parent content handler
100             * 
101             * @throws SAXException
102             *             See {@link ContentHandler}
103             */
104            public void endDocument() throws SAXException {
105                    m_parent.endDocument();
106                    m_locator = null;
107            }
108    
109            /**
110             * Proxies to the parent ContentHandler.
111             * 
112             * @param uri
113             *            See {@link ContentHandler}
114             * @param localName
115             *            See {@link ContentHandler}
116             * @param qName
117             *            See {@link ContentHandler}
118             * 
119             * @throws SAXException
120             *             See {@link ContentHandler}
121             */
122            public void endElement(String uri, String localName, String qName)
123                            throws SAXException {
124                    m_parent.endElement(uri, localName, qName);
125            }
126    
127            /**
128             * Proxies this call to the parent content handler
129             * 
130             * @param prefix
131             *            See {@link ContentHandler}
132             * 
133             * @throws SAXException
134             *             See {@link ContentHandler}
135             */
136            public void endPrefixMapping(String prefix) throws SAXException {
137                    m_parent.endPrefixMapping(prefix);
138            }
139    
140            /**
141             * Proxies this call to the parent content handler
142             * 
143             * @param ch
144             *            See {@link ContentHandler}
145             * @param start
146             *            See {@link ContentHandler}
147             * @param length
148             *            See {@link ContentHandler}
149             * 
150             * @throws SAXException
151             *             See {@link ContentHandler}
152             */
153            public void ignorableWhitespace(char[] ch, int start, int length)
154                            throws SAXException {
155                    m_parent.ignorableWhitespace(ch, start, length);
156            }
157    
158            /**
159             * Proxies this call to the parent content handler
160             * 
161             * @param target
162             *            {@link ContentHandler}
163             * @param data
164             *            See {@link ContentHandler}
165             * 
166             * @throws SAXException
167             *             See {@link ContentHandler}
168             */
169            public void processingInstruction(String target, String data)
170                            throws SAXException {
171                    m_parent.processingInstruction(target, data);
172            }
173    
174            /**
175             * Proxies this call to the parent content handler
176             * 
177             * @param name
178             *            {@link ContentHandler}
179             * 
180             * @throws SAXException
181             *             See {@link ContentHandler}
182             */
183            public void skippedEntity(String name) throws SAXException {
184                    m_parent.skippedEntity(name);
185            }
186    
187            /**
188             * Proxies this call to the parent content handler
189             * 
190             * @throws SAXException
191             *             See {@link ContentHandler}
192             */
193            public void startDocument() throws SAXException {
194                    // Locator for parse exceptions
195                    m_locator = null;
196    
197                    m_parent.startDocument();
198            }
199    
200            /**
201             * Detects the start of a <code>fbf:Transform</code> element, or proxies
202             * to the parent ContentHandler
203             * 
204             * @param uri
205             *            See {@link ContentHandler}
206             * @param localName
207             *            See {@link ContentHandler}
208             * @param qName
209             *            See {@link ContentHandler}
210             * @param atts
211             *            See {@link ContentHandler}
212             * 
213             * @throws SAXException
214             *             See {@link ContentHandler}
215             */
216            public void startElement(String uri, String localName, String qName,
217                            Attributes atts) throws SAXException {
218                    m_parent.startElement(uri, localName, qName, atts);
219            }
220    
221            /**
222             * Proxies this call to the parent content handler
223             * 
224             * @param prefix
225             *            See {@link ContentHandler}
226             * @param uri
227             *            See {@link ContentHandler}
228             * 
229             * @throws SAXException
230             *             See {@link ContentHandler}
231             */
232            public void startPrefixMapping(String prefix, String uri)
233                            throws SAXException {
234                    m_parent.startPrefixMapping(prefix, uri);
235            }
236    
237            /**
238             * Throws a parse exception
239             * 
240             * @param message
241             *            the exception message
242             * 
243             * @throws SAXParseException
244             *             always!
245             */
246            protected void parseException(String message) throws SAXParseException {
247                    throw new SAXParseException(message, m_locator);
248            }
249    
250            /**
251             * Throws a parse exception
252             * 
253             * @param message
254             *            the exception message
255             * @param cause
256             *            root cause
257             * 
258             * @throws SAXParseException
259             *             always!
260             */
261            protected void parseException(String message, Exception cause)
262                            throws SAXParseException {
263                    throw new SAXParseException(message, m_locator, cause);
264            }
265    }
