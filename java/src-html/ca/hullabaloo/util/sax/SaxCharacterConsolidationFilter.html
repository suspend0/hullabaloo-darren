001    package ca.hullabaloo.util.sax;
002    
003    import org.xml.sax.Attributes;
004    import org.xml.sax.ContentHandler;
005    import org.xml.sax.SAXException;
006    import org.xml.sax.helpers.XMLFilterImpl;
007    
008    import ca.hullabaloo.util.CharArrayBuffer;
009    
010    /**
011     * Consolidates consecutive {@link XMLFilterImpl#characters(char[],int,int)}
012     * into a single characters call. This easing writing of filters which further
013     * process the text.
014     * 
015     * @author Darren Gilroy
016     */
017    public class SaxCharacterConsolidationFilter extends XMLFilterImpl {
018            /** the current content we want to parse */
019            private final CharArrayBuffer m_buf = new CharArrayBuffer();
020    
021            /** which nodes to consolidate character events for */
022            private final ElementSelector m_selector;
023    
024            /**
025             * Creates a new AbstractSaxTagExpander that expands escaped tags of
026             * <code>tagName</code>
027             * 
028             * @param selector
029             *            only processes text when {@link ElementSelector#isSelected()}
030             *            is true.
031             */
032            public SaxCharacterConsolidationFilter(ElementSelector selector) {
033                    super();
034                    m_selector = selector;
035            }
036    
037            /**
038             * @see ContentHandler
039             */
040            public final void characters(char[] ch, int start, int length)
041                            throws SAXException {
042                    if (m_selector.isSelected()) {
043                            m_buf.append(ch, start, length);
044                    } else {
045                            super.characters(ch, start, length);
046                    }
047            }
048    
049            /**
050             * @see ContentHandler
051             */
052            public void endElement(String uri, String localName, String qName)
053                            throws SAXException {
054                    if (m_selector.isSelected()) {
055                            sendBuffer();
056                    }
057    
058                    super.endElement(uri, localName, qName);
059            }
060    
061            /**
062             * Resets internal state.
063             * 
064             * @throws SAXException
065             *             per api
066             */
067            public void startDocument() throws SAXException {
068                    m_buf.setLength(0);
069                    super.startDocument();
070            }
071    
072            /**
073             * @see org.xml.sax.ContentHandler#startElement(java.lang.String,
074             *      java.lang.String, java.lang.String, org.xml.sax.Attributes)
075             */
076            public final void startElement(String uri, String localName, String qName,
077                            Attributes atts) throws SAXException {
078                    if (m_selector.isSelected()) {
079                            sendBuffer();
080                    }
081    
082                    super.startElement(uri, localName, qName, atts);
083            }
084    
085            /**
086             * Sends the buffer as a single character event.
087             * 
088             * @throws SAXException
089             *             per api
090             */
091            private void sendBuffer() throws SAXException {
092                    int len = m_buf.length();
093    
094                    if (len > 0) {
095                            super.characters(m_buf.array(), 0, len);
096                            m_buf.setLength(0);
097                    }
098            }
099    }
