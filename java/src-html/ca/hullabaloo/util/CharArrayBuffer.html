001    package ca.hullabaloo.util;
002    
003    /**
004     * Much like a StringBuffer, except unsynchronized, gives access to the internal
005     * (possibly transient) char array, doesn't support insert(), has a larger
006     * default size, and throws a NullPointerException if you try to append a null
007     * value.
008     * 
009     * @author Darren Gilroy
010     */
011    public class CharArrayBuffer {
012            /** chunk size */
013            private static final int s_chunkSize = 512;
014    
015            /** the characters */
016            private char[] m_value;
017    
018            /** the number of characters used */
019            private int m_size;
020    
021            /**
022             * This no-arg constructor.
023             */
024            public CharArrayBuffer() {
025                    m_value = new char[s_chunkSize];
026            }
027    
028            /**
029             * Resets the number of characters in the buffer
030             * 
031             * @param newLength
032             *            the new length
033             * 
034             * @throws StringIndexOutOfBoundsException
035             *             if the new length is less than zero
036             */
037            public void setLength(int newLength) {
038                    if (newLength < 0) {
039                            throw new StringIndexOutOfBoundsException(newLength);
040                    }
041    
042                    if (newLength == 0) {
043                            if (m_value.length > s_chunkSize) {
044                                    m_value = null;
045                                    m_value = new char[s_chunkSize];
046                            }
047                    }
048    
049                    if (newLength > m_value.length) {
050                            grow(newLength);
051                    }
052    
053                    if (m_size < newLength) {
054                            for (; m_size < newLength; m_size++)
055                                    m_value[m_size] = '\0';
056                    } else {
057                            m_size = newLength;
058                    }
059            }
060    
061            /**
062             * Appends the data in the passed char array to the buffer
063             * 
064             * @param ch
065             *            the data to append
066             * 
067             * @return <code>this</code>
068             */
069            public CharArrayBuffer append(char[] ch) {
070                    int newSize = m_size + ch.length;
071    
072                    if (newSize > ch.length) {
073                            grow(newSize);
074                    }
075    
076                    System.arraycopy(ch, 0, m_value, m_size, ch.length);
077                    m_size = newSize;
078    
079                    return this;
080            }
081    
082            /**
083             * Appends the data in the passed char array to the buffer
084             * 
085             * @param ch
086             *            the data to append
087             * @param start
088             *            the position in the start data to start
089             * @param len
090             *            the number of chars to append
091             * 
092             * @return <code>this</code>
093             */
094            public CharArrayBuffer append(char[] ch, int start, int len) {
095                    int newSize = m_size + len;
096    
097                    if (newSize > m_value.length) {
098                            grow(newSize);
099                    }
100    
101                    System.arraycopy(ch, start, m_value, m_size, len);
102                    m_size = newSize;
103    
104                    return this;
105            }
106    
107            /**
108             * Appends the passed string to the buffer
109             * 
110             * @param str
111             *            the string to append
112             * 
113             * @return <code>this</code>
114             */
115            public CharArrayBuffer append(String str) {
116                    int len = str.length();
117    
118                    if (len > 0) {
119                            int newSize = m_size + len;
120    
121                            if (newSize > m_value.length) {
122                                    grow(newSize);
123                            }
124    
125                            str.getChars(0, len, m_value, m_size);
126                            m_size = newSize;
127                    }
128    
129                    return this;
130            }
131    
132            /**
133             * This is the underlying array at the point-in-time you call this method.
134             * Do not store a reference to this, it could be replaced by a call to
135             * append() or setLength() or other methods which may modify the content of
136             * the buffer. Modifications to this array affect the source CharArrayBuffer
137             * for as long as the CharArrayBuffer is using this array. The application
138             * must take care to not modify positions greater than
139             * CharArrayBuffer.length().
140             * 
141             * <p>
142             * <b>Note </b> that <code>array().length</code> may not, and probably
143             * will not, equal <code>length()</code>
144             * </p>
145             * 
146             * @return the CharArrayBuffer's underlying character array.
147             */
148            public char[] array() {
149                    return m_value;
150            }
151    
152            /**
153             * Returns the first location of the character array matched
154             * 
155             * @param search
156             *            the string to search for
157             * 
158             * @return the found position.
159             */
160            public int indexOf(char[] search) {
161                    return indexOf(search, 0);
162            }
163    
164            /**
165             * Returns the first location of the character array matched after the
166             * specified index
167             * 
168             * @param search
169             *            the string to search for
170             * @param fromIndex
171             *            the position in the buffer to start the search
172             * 
173             * @return the position, or -1 if not found
174             */
175            public int indexOf(char[] search, int fromIndex) {
176                    return indexOf(m_value, 0, m_size, search, 0, search.length, fromIndex);
177            }
178    
179            /**
180             * This code is Copyright Sun Microsystems and distributed as part of
181             * java.lang.String
182             * 
183             * <p>
184             * Code shared by String and StringBuffer to do searches. The source is the
185             * character array being searched, and the target is the string being
186             * searched for.
187             * </p>
188             * 
189             * @param source
190             *            the characters being searched.
191             * @param sourceOffset
192             *            offset of the source string.
193             * @param sourceCount
194             *            count of the source string.
195             * @param target
196             *            the characters being searched for.
197             * @param targetOffset
198             *            offset of the target string.
199             * @param targetCount
200             *            count of the target string.
201             * @param fromIndex
202             *            the index to begin searching from.
203             * 
204             * @return found position, or -1 if not found
205             */
206            public static int indexOf(char[] source, int sourceOffset, int sourceCount,
207                            char[] target, int targetOffset, int targetCount, int fromIndex) {
208                    if (fromIndex >= sourceCount) {
209                            return ((targetCount == 0) ? sourceCount : (-1));
210                    }
211    
212                    if (fromIndex < 0) {
213                            fromIndex = 0;
214                    }
215    
216                    if (targetCount == 0) {
217                            return fromIndex;
218                    }
219    
220                    char first = target[targetOffset];
221                    int max = sourceOffset + (sourceCount - targetCount);
222    
223                    for (int i = sourceOffset + fromIndex; i <= max; i++) {
224                            /* Look for first character. */
225                            if (source[i] != first) {
226                                    while ((++i <= max) && (source[i] != first))
227                                            ;
228                            }
229    
230                            /* Found first character, now look at the rest of v2 */
231                            if (i <= max) {
232                                    int j = i + 1;
233                                    int end = (j + targetCount) - 1;
234    
235                                    for (int k = targetOffset + 1; (j < end)
236                                                    && (source[j] == target[k]); j++, k++)
237                                            ;
238    
239                                    if (j == end) {
240                                            /* Found whole string. */
241                                            return i - sourceOffset;
242                                    }
243                            }
244                    }
245    
246                    return -1;
247            }
248    
249            /**
250             * Returns the number of characters in the buffer.
251             * 
252             * @return length
253             */
254            public int length() {
255                    return m_size;
256            }
257    
258            /**
259             * a string representation
260             * 
261             * @return the char array as a string.
262             */
263            public String toString() {
264                    return new String(m_value, 0, m_size);
265            }
266    
267            /**
268             * Expands the underlying character array to the passe size
269             * 
270             * @param minSize
271             *            the new minimum size
272             */
273            private void grow(int minSize) {
274                    // find the next multiple of the chunk size greater than the minSize,
275                    // or double, whichever is greater
276                    int newSize = (minSize + s_chunkSize) - (minSize % s_chunkSize);
277                    newSize = Math.max(newSize, m_value.length * 2);
278    
279                    // grow the array
280                    char[] newValue = new char[newSize];
281                    System.arraycopy(m_value, 0, newValue, 0, m_size);
282                    m_value = newValue;
283            }
284    }
